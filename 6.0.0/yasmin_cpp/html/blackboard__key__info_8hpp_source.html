76 template <typename T>
77 BlackboardKeyInfo(std::string key_name, std::string key_description, T value)
78 : name(std::move(key_name)), description(std::move(key_description)),
79 has_default(true), default_value(std::make_shared<std::decay_t<T>>(
80 std::forward<T>(value))),
81 default_value_type(demangle_type(typeid(std::decay_t<T>).name())) {
82 using ValueType = std::decay_t<T>;
83 auto stored = std::static_pointer_cast<ValueType>(this->default_value);
84 this->inject_default = [stored](Blackboard &bb, const std::string &key) {
85 bb.set<ValueType>(key, *stored);
86 };
87 }

96 BlackboardKeyInfo(std::string key_name, std::string key_description,
97 const char *value)
98 : name(std::move(key_name)), description(std::move(key_description)),
99 has_default(true), default_value(std::make_shared<std::string>(value)),
100 default_value_type(demangle_type(typeid(std::string).name())) {
101 auto stored = std::static_pointer_cast<std::string>(this->default_value);
102 this->inject_default = [stored](Blackboard &bb, const std::string &key) {
103 bb.set<std::string>(key, *stored);
104 };
