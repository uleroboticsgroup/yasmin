96 template <class T> void set(const std::string &name, T value) {
97 using DecayT = std::decay_t<T>;
98
99 YASMIN_LOG_DEBUG("Setting '%s' in the blackboard", name.c_str());
100
101 std::lock_guard<std::recursive_mutex> lk(this->storage->mutex);
102
103 // Apply remapping if exists
104 const std::string &key = this->remap(name);
105 const std::string type_name = demangle_type(typeid(DecayT).name());
106
107 auto type_it = this->storage->type_registry.find(key);
108 if (type_it != this->storage->type_registry.end() &&
109 type_it->second == type_name) {
110 // Same type: update existing value in-place (avoids allocation)
111 *(std::static_pointer_cast<DecayT>(this->storage->values.at(key))) =
112 std::move(value);
113 } else {
114 // New key or different type: (re)create entry
115 this->storage->values[key] = std::make_shared<DecayT>(std::move(value));
116 this->storage->type_registry[key] = type_name;
117 }
118 }

127 template <class T> T get(const std::string &key) const {
128
129 YASMIN_LOG_DEBUG("Getting '%s' from the blackboard", key.c_str());
130
131 std::lock_guard<std::recursive_mutex> lk(this->storage->mutex);
132
133 const std::string &remapped_key = this->remap(key);
134 auto it = this->storage->values.find(remapped_key);
135 if (it == this->storage->values.end()) {
136 throw std::runtime_error("Element '" + key +
137 "' does not exist in the blackboard");
138 }
139
140 // Return the value casted to the requested type
141 return *(std::static_pointer_cast<T>(it->second));
142 }
