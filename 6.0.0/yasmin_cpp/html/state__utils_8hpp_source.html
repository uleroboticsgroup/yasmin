36inline Outcomes generate_possible_outcomes(const OutcomeMap &outcome_map,
37 const std::string &default_outcome) {
38 Outcomes possible_outcomes;
39 possible_outcomes.insert(default_outcome);
40
41 for (const auto &[outcome, requirements] : outcome_map) {
42 (void)requirements;
43 possible_outcomes.insert(outcome);
44 }
45
46 return possible_outcomes;

55inline std::string join_outcomes(const Outcomes &items,

57 std::string result;

117template <typename LookupFn>
118Outcomes evaluate_satisfied_outcomes(const OutcomeMap &outcome_map,
119 LookupFn get_actual) {
120 Outcomes satisfied_outcomes;
121
122 for (const auto &[outcome, requirements] : outcome_map) {
123 bool satisfied = true;
124
125 for (const auto &[state_name, expected_outcome] : requirements) {
126 if (get_actual(state_name) != expected_outcome) {
127 satisfied = false;
128 break;
129 }
130 }
131
132 if (satisfied) {
133 satisfied_outcomes.insert(outcome);
134 }
135 }
136
137 return satisfied_outcomes;
138}

148inline std::string resolve_outcome(const Outcomes &satisfied_outcomes,
149 const std::string &default_outcome,
150 const std::string &context_name = "") {
151 if (satisfied_outcomes.empty()) {
152 return default_outcome;
153 }
154
155 if (satisfied_outcomes.size() > 1) {
156 throw std::logic_error("More than one satisfied outcomes (" +
157 join_outcomes(satisfied_outcomes) + ") after " +
158 context_name);
159 }
160
161 return *satisfied_outcomes.begin();
162}

170inline void
171apply_parameter_mappings(const std::string &container_type,
172 const ParameterMappingsMap &parameter_mappings,
173 const std::string &state_name, State &parent,
174 const State::SharedPtr &child) {
175
176 const auto mappings_it = parameter_mappings.find(state_name);
177 if (mappings_it == parameter_mappings.end()) {
178 return;
179 }
180
181 for (const auto &[child_parameter, parent_parameter] : mappings_it->second) {
182 if (!parent.is_parameter_declared(parent_parameter)) {
183 throw std::runtime_error(
184 container_type + " parameter '" + parent_parameter +
185 "' is not declared while configuring child state '" + state_name +
186 "'");
187 }
188
189 if (!parent.has_parameter(parent_parameter)) {
190 throw std::runtime_error(
191 container_type + " parameter '" + parent_parameter +
192 "' has no value while configuring child state '" + state_name + "'");
193 }
194
195 if (!child->is_parameter_declared(child_parameter)) {
196 throw std::runtime_error("Child state '" + state_name +
197 "' does not declare parameter '" +
198 child_parameter + "'");
199 }
200
201 child->copy_parameter_from(parent, parent_parameter, child_parameter);
202 }
203}
