98 static std::vector<uint8_t> py_buffer_to_vector(const py::handle &value) {
99 char *data = nullptr;
100 py::ssize_t size = 0;
101
102 if (PyBytes_Check(value.ptr()) != 0) {
103 if (PyBytes_AsStringAndSize(value.ptr(), &data, &size) != 0) {
104 throw py::error_already_set();
105 }
106 } else if (PyByteArray_Check(value.ptr()) != 0) {
107 data = PyByteArray_AsString(value.ptr());
108 size = PyByteArray_Size(value.ptr());
109 } else {
110 throw std::runtime_error("Expected bytes or bytearray object");
111 }
112
113 return std::vector<uint8_t>(reinterpret_cast<uint8_t *>(data),
114 reinterpret_cast<uint8_t *>(data) + size);
115 }

201 template <typename T>
202 static std::vector<T> sequence_to_vector(const py::sequence &seq) {
203 std::vector<T> result;
204 result.reserve(static_cast<std::size_t>(py::len(seq)));
205
206 for (auto item : seq) {
207 result.push_back(py::cast<T>(item));
208 }
209
210 return result;
211 }

250 template <typename T>
251 static std::unordered_map<std::string, T>
252 dict_to_unordered_map(const py::dict &dict) {
253 std::unordered_map<std::string, T> result;
254 result.reserve(static_cast<std::size_t>(py::len(dict)));
255
256 for (auto item : dict) {
257 result.emplace(py::cast<std::string>(item.first),
258 py::cast<T>(item.second));
259 }
260
261 return result;
262 }

315 void set(const std::string &key, py::object value) {
316 if (py::isinstance<py::bool_>(value)) {
317 this->blackboard->set<bool>(key, value.cast<bool>());
318 return;
319 }
320
321 if (is_python_int_like(value)) {
322 this->blackboard->set<std::int64_t>(key, value.cast<std::int64_t>());
323 return;
324 }
325
326 if (py::isinstance<py::float_>(value)) {
327 this->blackboard->set<double>(key, value.cast<double>());
328 return;
329 }
330
331 if (is_python_bytes_like(value)) {
332 this->blackboard->set<std::vector<uint8_t>>(key,
333 py_buffer_to_vector(value));
334 return;
335 }
336
337 if (py::isinstance<py::str>(value)) {
338 this->blackboard->set<std::string>(key, value.cast<std::string>());
339 return;
340 }
341
342 if (yasmin::callback_signal_pyutils::is_python_callback_signal_like(
343 value)) {
344 this->blackboard->set<yasmin::CallbackSignal::SharedPtr>(
345 key,
346 yasmin::callback_signal_pyutils::cast_python_callback_signal(value));
347 return;
348 }
349
350 if (is_python_sequence_like(value)) {
351 py::sequence seq = value.cast<py::sequence>();
352
353 // Empty sequences do not carry enough type information to infer a native
354 // element type safely.
355 if (py::len(seq) == 0) {
356 this->blackboard->set<py::object>(key, value);
357 return;
358 }
359
360 // Bool must be checked before int because Python bool is also an int.
361 if (sequence_matches(seq, [](const py::handle &item) {
362 return py::isinstance<py::bool_>(item);
363 })) {
364 this->blackboard->set<BoolVector>(key, sequence_to_vector<bool>(seq));
365 return;
366 }
367
368 if (sequence_matches(seq, [](const py::handle &item) {
369 return is_python_int_like(item);
370 })) {
371 this->blackboard->set<IntVector>(key,
372 sequence_to_vector<std::int64_t>(seq));
373 return;
374 }
375
376 // Mixed int/float sequences are normalized to double.
377 if (sequence_matches(seq, [](const py::handle &item) {
378 return is_python_number_like(item);
379 })) {
380 this->blackboard->set<FloatVector>(key,
381 sequence_to_vector<double>(seq));
382 return;
383 }
384
385 if (sequence_matches(seq, [](const py::handle &item) {
386 return py::isinstance<py::str>(item);
387 })) {
388 this->blackboard->set<StringVector>(
389 key, sequence_to_vector<std::string>(seq));
390 return;
391 }
392
393 this->blackboard->set<py::object>(key, value);
394 return;
395 }
396
397 if (py::isinstance<py::dict>(value)) {
398 py::dict dict = value.cast<py::dict>();
399
400 // Only dict[str, T] with a homogeneous value type is converted into a
401 // native unordered_map.
402 if (py::len(dict) == 0 || !dict_has_only_string_keys(dict)) {
403 this->blackboard->set<py::object>(key, value);
404 return;
405 }
406
407 if (dict_values_match(dict, [](const py::handle &item) {
408 return py::isinstance<py::bool_>(item);
409 })) {
410 this->blackboard->set<BoolDict>(key, dict_to_unordered_map<bool>(dict));
411 return;
412 }
413
414 if (dict_values_match(dict, [](const py::handle &item) {
415 return is_python_int_like(item);
416 })) {
417 this->blackboard->set<IntDict>(
418 key, dict_to_unordered_map<std::int64_t>(dict));
419 return;
420 }
421
422 // Mixed int/float dictionaries are normalized to double.
423 if (dict_values_match(dict, [](const py::handle &item) {
424 return is_python_number_like(item);
425 })) {
426 this->blackboard->set<FloatDict>(key,
427 dict_to_unordered_map<double>(dict));
428 return;
429 }
430
431 if (dict_values_match(dict, [](const py::handle &item) {
432 return py::isinstance<py::str>(item);
433 })) {
434 this->blackboard->set<StringDict>(
435 key, dict_to_unordered_map<std::string>(dict));
436 return;
437 }
438
439 this->blackboard->set<py::object>(key, value);
440 return;
441 }
442
443 this->blackboard->set<py::object>(key, value);
444 }

454 py::object get(const std::string &key) const {
455 const std::string type = this->blackboard->get_type(key);
456
457 // py::object is stored directly without extra wrapping
458 if (type == demangle_type(typeid(py::object).name())) {
459 return this->blackboard->get<py::object>(key);
460 }
461
462 using Getter =
463 std::function<py::object(const Blackboard &, const std::string &)>;
464
465 static const auto *const GETTERS = []() {
466 using Map = std::unordered_map<std::string, Getter>;
467 auto *m = new Map();
468
469 auto add = [&](auto dummy) {
470 using T = decltype(dummy);
471 (*m)[demangle_type(typeid(T).name())] = [](const Blackboard &bb,
472 const std::string &k) {
473 return py::cast(bb.get<T>(k));
474 };
475 };
476
477 auto add_bytes = [&](auto dummy) {
478 using T = decltype(dummy);
479 (*m)[demangle_type(typeid(T).name())] = [](const Blackboard &bb,
480 const std::string &k) {
481 return byte_vector_to_py_bytes(bb.get<T>(k));
482 };
483 };
484
485 // Scalars
486 add(std::string{});
487 add(std::int64_t{});
488 add(int{});
489 add(long{});
490 add(float{});
491 add(double{});
492 add(bool{});
493 add(yasmin::CallbackSignal::SharedPtr{});
494
495 // Vectors
496 add(StringVector{});
497 add(IntVector{});
498 add(FloatVector{});
499 add(BoolVector{});
500 // Alternate C++ integer widths that pybind11 may store
501 add(std::vector<int>{});
502 add(std::vector<long>{});
503 add(std::vector<long long>{});
504 add(std::vector<float>{});
505 // Byte vectors
506 add_bytes(std::vector<uint8_t>{});
507 add_bytes(std::vector<unsigned char>{});
508 add_bytes(std::vector<char>{});
509
510 // Dicts
511 add(StringDict{});
512 add(IntDict{});
513 add(FloatDict{});
514 add(BoolDict{});
515 add(std::unordered_map<std::string, int>{});
516 add(std::unordered_map<std::string, long>{});
517 add(std::unordered_map<std::string, long long>{});
518 add(std::unordered_map<std::string, float>{});
519
520 return m;
521 }();
522
523 auto it = GETTERS->find(type);
524 if (it != GETTERS->end()) {
525 return it->second(*this->blackboard, key);
526 }
527
528 return py::none();
529 }
