34template <typename InterfaceT>
35std::vector<uint8_t> serialize_interface(const InterfaceT &interface_value) {
36 rclcpp::Serialization<InterfaceT> serializer;
37 rclcpp::SerializedMessage serialized_message;
38 serializer.serialize_message(&interface_value, &serialized_message);
39
40 const auto &rcl_serialized_message =
41 serialized_message.get_rcl_serialized_message();
42
43 return std::vector<uint8_t>(rcl_serialized_message.buffer,
44 rcl_serialized_message.buffer +
45 rcl_serialized_message.buffer_length);
46}

56template <typename InterfaceT>
57InterfaceT deserialize_interface(const std::vector<uint8_t> &serialized_data) {
58 rclcpp::Serialization<InterfaceT> serializer;
59 rclcpp::SerializedMessage serialized_message(serialized_data.size());
60
61 auto &rcl_serialized_message =
62 serialized_message.get_rcl_serialized_message();
63
64 if (!serialized_data.empty()) {
65 std::memcpy(rcl_serialized_message.buffer, serialized_data.data(),
66 serialized_data.size());
67 }
68
69 rcl_serialized_message.buffer_length = serialized_data.size();
70
71 InterfaceT interface_value;
72 serializer.deserialize_message(&serialized_message, &interface_value);
73
74 return interface_value;
75}
