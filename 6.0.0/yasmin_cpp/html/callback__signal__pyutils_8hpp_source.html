40inline bool is_python_callback_signal_like(const py::handle &value) {
41 if (value.is_none()) {
42 return false;
43 }
44
45 try {
46#if PYBIND11_VERSION_MAJOR > 2 || \
47 (PYBIND11_VERSION_MAJOR == 2 && PYBIND11_VERSION_MINOR >= 6)
48 py::module_ callback_signal_module =
49 py::module_::import("yasmin.callback_signal");
50#else
51 py::module callback_signal_module =
52 py::module::import("yasmin.callback_signal");
53#endif
54 py::object callback_signal_class =
55 callback_signal_module.attr("CallbackSignal");
56 const int is_instance =
57 PyObject_IsInstance(value.ptr(), callback_signal_class.ptr());
58
59 if (is_instance == 1) {
60 return true;
61 }
62
63 if (is_instance == -1) {
64 PyErr_Clear();
65 }
66 } catch (const py::error_already_set &) {
67 PyErr_Clear();
68 }
69
70 return false;
71}
