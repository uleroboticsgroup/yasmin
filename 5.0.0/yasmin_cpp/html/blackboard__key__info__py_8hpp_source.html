49inline std::vector<uint8_t> py_buffer_to_vector(const py::handle &value) {
50 char *data = nullptr;
51 py::ssize_t size = 0;
52
53 if (PyBytes_Check(value.ptr()) != 0) {
54 if (PyBytes_AsStringAndSize(value.ptr(), &data, &size) != 0) {
55 throw py::error_already_set();
56 }
57 } else if (PyByteArray_Check(value.ptr()) != 0) {
58 data = PyByteArray_AsString(value.ptr());
59 size = PyByteArray_Size(value.ptr());
60 } else {
61 throw std::runtime_error("Expected bytes or bytearray object");
62 }
63
64 return std::vector<uint8_t>(reinterpret_cast<uint8_t *>(data),
65 reinterpret_cast<uint8_t *>(data) + size);
66}

94template <typename Predicate>
95inline bool sequence_matches(const py::sequence &seq, Predicate pred) {
96 for (auto item : seq) {

104template <typename T>
105inline std::vector<T> sequence_to_vector(const py::sequence &seq) {
106 std::vector<T> result;
107 result.reserve(static_cast<std::size_t>(py::len(seq)));
108
109 for (auto item : seq) {
110 result.push_back(py::cast<T>(item));
111 }
112
113 return result;
114}

116inline bool dict_has_only_string_keys(const py::dict &dict) {
117 for (auto item : dict) {
118 if (!py::isinstance<py::str>(item.first)) {
119 return false;
120 }

122 return true;

125template <typename Predicate>
126inline bool dict_values_match(const py::dict &dict, Predicate pred) {
127 for (auto item : dict) {
128 if (!pred(item.second)) {
129 return false;
130 }
131 }
132 return true;

135template <typename T>
136inline std::unordered_map<std::string, T>
137dict_to_unordered_map(const py::dict &dict) {
138 std::unordered_map<std::string, T> result;
139 result.reserve(static_cast<std::size_t>(py::len(dict)));
140
141 for (auto item : dict) {
142 result.emplace(py::cast<std::string>(item.first), py::cast<T>(item.second));
143 }
144
145 return result;
146}

161inline BlackboardKeyInfo
162blackboard_key_info_from_pyobject(const std::string &key_name,
163 py::object value) {
164 using namespace detail;
165
166 if (py::isinstance<py::bool_>(value)) {
167 return BlackboardKeyInfo(key_name, "", value.cast<bool>());
168 }
169
170 if (is_python_int_like(value)) {
171 return BlackboardKeyInfo(key_name, "", value.cast<std::int64_t>());
172 }
173
174 if (py::isinstance<py::float_>(value)) {
175 return BlackboardKeyInfo(key_name, "", value.cast<double>());
176 }
177
178 if (is_python_bytes_like(value)) {
179 return BlackboardKeyInfo(key_name, "", py_buffer_to_vector(value));
180 }
181
182 if (py::isinstance<py::str>(value)) {
183 return BlackboardKeyInfo(key_name, "", value.cast<std::string>());
184 }
185
186 if (is_python_sequence_like(value)) {
187 py::sequence seq = value.cast<py::sequence>();
188
189 if (py::len(seq) == 0) {
190 BlackboardKeyInfo info(key_name);
191 info.has_default = true;
192 info.default_value_type = "py::object";
193 auto stored = std::make_shared<py::object>(value);
194 info.default_value = stored;
195 info.inject_default = [stored](Blackboard &bb, const std::string &key) {
196 py::gil_scoped_acquire gil;
197 bb.set<py::object>(key, *stored);
198 };
199 return info;
200 }
201
202 if (sequence_matches(seq, [](const py::handle &item) {
203 return py::isinstance<py::bool_>(item);
204 })) {
205 return BlackboardKeyInfo(key_name, "", sequence_to_vector<bool>(seq));
206 }
207
208 if (sequence_matches(seq, [](const py::handle &item) {
209 return is_python_int_like(item);
210 })) {
211 return BlackboardKeyInfo(key_name, "",
212 sequence_to_vector<std::int64_t>(seq));
213 }
214
215 if (sequence_matches(seq, [](const py::handle &item) {
216 return is_python_number_like(item);
217 })) {
218 return BlackboardKeyInfo(key_name, "", sequence_to_vector<double>(seq));
219 }
220
221 if (sequence_matches(seq, [](const py::handle &item) {
222 return py::isinstance<py::str>(item);
223 })) {
224 return BlackboardKeyInfo(key_name, "",
225 sequence_to_vector<std::string>(seq));
226 }
227 }
228
229 if (py::isinstance<py::dict>(value)) {
230 py::dict dict = value.cast<py::dict>();
231
232 if (py::len(dict) != 0 && dict_has_only_string_keys(dict)) {
233 if (dict_values_match(dict, [](const py::handle &item) {
234 return py::isinstance<py::bool_>(item);
235 })) {
236 return BlackboardKeyInfo(key_name, "",
237 dict_to_unordered_map<bool>(dict));
238 }
239
240 if (dict_values_match(dict, [](const py::handle &item) {
241 return is_python_int_like(item);
242 })) {
243 return BlackboardKeyInfo(key_name, "",
244 dict_to_unordered_map<std::int64_t>(dict));
245 }
246
247 if (dict_values_match(dict, [](const py::handle &item) {
248 return is_python_number_like(item);
249 })) {
250 return BlackboardKeyInfo(key_name, "",
251 dict_to_unordered_map<double>(dict));
252 }
253
254 if (dict_values_match(dict, [](const py::handle &item) {
255 return py::isinstance<py::str>(item);
256 })) {
257 return BlackboardKeyInfo(key_name, "",
258 dict_to_unordered_map<std::string>(dict));
259 }
260 }
261 }
262
263 BlackboardKeyInfo info(key_name);
264 info.has_default = true;
265 info.default_value_type = "py::object";
266
267 auto stored = std::make_shared<py::object>(value);
268 info.default_value = stored;
269 info.inject_default = [stored](Blackboard &bb, const std::string &key) {
270 py::gil_scoped_acquire gil;
271 bb.set<py::object>(key, *stored);
272 };
273
274 return info;
275}

283inline py::object
284blackboard_key_info_get_py_default(const BlackboardKeyInfo &info) {
285 using namespace detail;
286
287 if (!info.has_default) {
288 return py::none();
289 }
290
291 const std::string &type = info.default_value_type;
292
293 if (is_exact_cpp_type<std::vector<uint8_t>>(type)) {
294 return byte_vector_to_py_bytes(
295 info.get_default_value<std::vector<uint8_t>>());
296 }
297 if (is_exact_cpp_type<std::vector<unsigned char>>(type)) {
298 return byte_vector_to_py_bytes(
299 info.get_default_value<std::vector<unsigned char>>());
300 }
301 if (is_exact_cpp_type<std::vector<char>>(type)) {
302 return byte_vector_to_py_bytes(info.get_default_value<std::vector<char>>());
303 }
304
305 if (is_exact_cpp_type<StringVector>(type)) {
306 return py::cast(info.get_default_value<StringVector>());
307 }
308 if (is_exact_cpp_type<IntVector>(type) ||
309 is_exact_cpp_type<std::vector<int>>(type) ||
310 is_exact_cpp_type<std::vector<long>>(type) ||
311 is_exact_cpp_type<std::vector<long long>>(type)) {
312 if (is_exact_cpp_type<IntVector>(type)) {
313 return py::cast(info.get_default_value<IntVector>());
314 }
315 if (is_exact_cpp_type<std::vector<int>>(type)) {
316 return py::cast(info.get_default_value<std::vector<int>>());
317 }
318 if (is_exact_cpp_type<std::vector<long>>(type)) {
319 return py::cast(info.get_default_value<std::vector<long>>());
320 }
321 return py::cast(info.get_default_value<std::vector<long long>>());
322 }
323 if (is_exact_cpp_type<FloatVector>(type) ||
324 is_exact_cpp_type<std::vector<float>>(type)) {
325 if (is_exact_cpp_type<FloatVector>(type)) {
326 return py::cast(info.get_default_value<FloatVector>());
327 }
328 return py::cast(info.get_default_value<std::vector<float>>());
329 }
330 if (is_exact_cpp_type<BoolVector>(type)) {
331 return py::cast(info.get_default_value<BoolVector>());
332 }
333
334 if (is_exact_cpp_type<StringDict>(type)) {
335 return py::cast(info.get_default_value<StringDict>());
336 }
337 if (is_exact_cpp_type<IntDict>(type) ||
338 is_exact_cpp_type<std::unordered_map<std::string, int>>(type) ||
339 is_exact_cpp_type<std::unordered_map<std::string, long>>(type) ||
340 is_exact_cpp_type<std::unordered_map<std::string, long long>>(type)) {
341 if (is_exact_cpp_type<IntDict>(type)) {
342 return py::cast(info.get_default_value<IntDict>());
343 }
344 if (is_exact_cpp_type<std::unordered_map<std::string, int>>(type)) {
345 return py::cast(
346 info.get_default_value<std::unordered_map<std::string, int>>());
347 }
348 if (is_exact_cpp_type<std::unordered_map<std::string, long>>(type)) {
349 return py::cast(
350 info.get_default_value<std::unordered_map<std::string, long>>());
351 }
352 return py::cast(
353 info.get_default_value<std::unordered_map<std::string, long long>>());
354 }
355 if (is_exact_cpp_type<FloatDict>(type) ||
356 is_exact_cpp_type<std::unordered_map<std::string, float>>(type)) {
357 if (is_exact_cpp_type<FloatDict>(type)) {
358 return py::cast(info.get_default_value<FloatDict>());
359 }
360 return py::cast(
361 info.get_default_value<std::unordered_map<std::string, float>>());
362 }
363 if (is_exact_cpp_type<BoolDict>(type)) {
364 return py::cast(info.get_default_value<BoolDict>());
365 }
366
367 if (is_exact_cpp_type<std::string>(type)) {
368 return py::cast(info.get_default_value<std::string>());
369 }
370 if (is_exact_cpp_type<std::int64_t>(type)) {
371 return py::cast(info.get_default_value<std::int64_t>());
372 }
373 if (is_exact_cpp_type<int>(type)) {
374 return py::cast(info.get_default_value<int>());
375 }
376 if (is_exact_cpp_type<long>(type)) {
377 return py::cast(info.get_default_value<long>());
378 }
379 if (is_exact_cpp_type<long long>(type)) {
380 return py::cast(info.get_default_value<long long>());
381 }
382 if (is_exact_cpp_type<float>(type)) {
383 return py::cast(info.get_default_value<float>());
384 }
385 if (is_exact_cpp_type<double>(type)) {
386 return py::cast(info.get_default_value<double>());
387 }
388 if (is_exact_cpp_type<bool>(type)) {
389 return py::cast(info.get_default_value<bool>());
390 }
391 if (type == "py::object") {
392 return info.get_default_value<py::object>();
393 }
394
395 throw std::runtime_error("Unsupported BlackboardKeyInfo default value type "
396 "for Python conversion: " +
397 type);
398}
