77 template <typename T>
78 BlackboardKeyInfo(std::string key_name, std::string key_description, T value)
79 : name(std::move(key_name)), description(std::move(key_description)),
80 has_default(true), default_value(std::make_shared<std::decay_t<T>>(
81 std::forward<T>(value))),
82 default_value_type(demangle_type(typeid(std::decay_t<T>).name())) {
83 using ValueType = std::decay_t<T>;
84 auto stored = std::static_pointer_cast<ValueType>(this->default_value);
85 inject_default = [stored](Blackboard &bb, const std::string &key) {
86 bb.set<ValueType>(key, *stored);
87 };
88 }
