64 template <typename ActionT>
65 static typename rclcpp_action::Client<ActionT>::SharedPtr
66 get_or_create_action_client(
67 rclcpp::Node::SharedPtr node, const std::string &action_name,
68 rclcpp::CallbackGroup::SharedPtr callback_group = nullptr) {
69
70 // Create a unique key
71 std::string node_name = node->get_name();
72 std::string action_type_name = get_type_name<ActionT>();
73 std::string callback_group_name = get_callback_group_name(callback_group);
74 auto cache_key =
75 std::make_tuple(node_name, action_type_name, action_name,
76 callback_group_name, std::type_index(typeid(ActionT)));
77
78 std::lock_guard<std::recursive_mutex> lock(get_lock());
79
80 // Check if action client already exists in cache
81 auto &action_clients = get_action_clients();
82 auto it = action_clients.find(cache_key);
83
84 if (it != action_clients.end()) {
85 YASMIN_LOG_INFO("Reusing existing action client for '%s' of type '%s'",
86 action_name.c_str(), action_type_name.c_str());
87 return std::static_pointer_cast<rclcpp_action::Client<ActionT>>(
88 it->second);
89 }
90
91 // Create new action client if not in cache
92 YASMIN_LOG_INFO("Creating new action client for '%s' of type '%s'",
93 action_name.c_str(), action_type_name.c_str());
94
95 auto action_client = rclcpp_action::create_client<ActionT>(
96 node, action_name, callback_group);
97
98 // Store in cache (as void pointer for type erasure)
99 action_clients[cache_key] = std::static_pointer_cast<void>(action_client);
100
101 return action_client;
102 }

143#if __has_include("rclcpp/version.h")
144#include "rclcpp/version.h"

146 auto qos = rclcpp::QoS(
147 rclcpp::QoSInitialization::from_rmw(rmw_qos_profile_services_default));
148#else // Humble and Iron
149 auto qos = rmw_qos_profile_services_default;
150#endif
151#else // Foxy and Galactic
152 auto qos = rmw_qos_profile_services_default;
153#endif
154
155 auto service_client =
156 node->create_client<ServiceT>(service_name, qos, callback_group);
157
158 // Store in cache (as void pointer for type erasure)
159 service_clients[cache_key] = std::static_pointer_cast<void>(service_client);
160
161 return service_client;

174 template <typename MsgT>
175 static typename rclcpp::Publisher<MsgT>::SharedPtr get_or_create_publisher(
176 rclcpp::Node::SharedPtr node, const std::string &topic_name,
177 const rclcpp::QoS &qos_profile = rclcpp::QoS(10),
178 rclcpp::CallbackGroup::SharedPtr callback_group = nullptr) {
179
180 // Create a unique key
181 std::string node_name = node->get_name();
182 std::string msg_type_name = get_type_name<MsgT>();
183 std::string qos_hash = hash_qos_profile(qos_profile);
184 auto cache_key = std::make_tuple(node_name, msg_type_name, topic_name,
185 qos_hash, std::type_index(typeid(MsgT)));
186
187 std::lock_guard<std::recursive_mutex> lock(get_lock());
188
189 // Check if publisher already exists in cache
190 auto &publishers = get_publishers();
191 auto it = publishers.find(cache_key);
192
193 if (it != publishers.end()) {
194 YASMIN_LOG_INFO("Reusing existing publisher for topic '%s' of type '%s'",
195 topic_name.c_str(), msg_type_name.c_str());
196 return std::static_pointer_cast<rclcpp::Publisher<MsgT>>(it->second);
197 }
198
199 // Create new publisher if not in cache
200 YASMIN_LOG_INFO("Creating new publisher for topic '%s' of type '%s'",
201 topic_name.c_str(), msg_type_name.c_str());
202
203 rclcpp::PublisherOptions options;
204 options.callback_group = callback_group;
205
206 auto publisher =
207 node->create_publisher<MsgT>(topic_name, qos_profile, options);
208
209 // Store in cache (as void pointer for type erasure)
210 publishers[cache_key] = std::static_pointer_cast<void>(publisher);
211
212 return publisher;
213 }
