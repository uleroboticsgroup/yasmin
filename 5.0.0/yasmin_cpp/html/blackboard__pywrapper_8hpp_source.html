96 static std::vector<uint8_t> py_buffer_to_vector(const py::handle &value) {
97 char *data = nullptr;
98 py::ssize_t size = 0;
99
100 if (PyBytes_Check(value.ptr()) != 0) {
101 if (PyBytes_AsStringAndSize(value.ptr(), &data, &size) != 0) {
102 throw py::error_already_set();
103 }
104 } else if (PyByteArray_Check(value.ptr()) != 0) {
105 data = PyByteArray_AsString(value.ptr());
106 size = PyByteArray_Size(value.ptr());
107 } else {
108 throw std::runtime_error("Expected bytes or bytearray object");
109 }
110
111 return std::vector<uint8_t>(reinterpret_cast<uint8_t *>(data),
112 reinterpret_cast<uint8_t *>(data) + size);
113 }

199 template <typename T>
200 static std::vector<T> sequence_to_vector(const py::sequence &seq) {
201 std::vector<T> result;
202 result.reserve(static_cast<std::size_t>(py::len(seq)));
203
204 for (auto item : seq) {
205 result.push_back(py::cast<T>(item));

248 template <typename T>
249 static std::unordered_map<std::string, T>
250 dict_to_unordered_map(const py::dict &dict) {
251 std::unordered_map<std::string, T> result;
252 result.reserve(static_cast<std::size_t>(py::len(dict)));
253
254 for (auto item : dict) {
255 result.emplace(py::cast<std::string>(item.first),
256 py::cast<T>(item.second));
257 }
258
259 return result;
260 }

303 void set(const std::string &key, py::object value) {
304 if (py::isinstance<py::bool_>(value)) {
305 this->blackboard->set<bool>(key, value.cast<bool>());
306 return;
307 }
308
309 if (is_python_int_like(value)) {
310 this->blackboard->set<std::int64_t>(key, value.cast<std::int64_t>());
311 return;
312 }
313
314 if (py::isinstance<py::float_>(value)) {
315 this->blackboard->set<double>(key, value.cast<double>());
316 return;
317 }
318
319 if (is_python_bytes_like(value)) {
320 this->blackboard->set<std::vector<uint8_t>>(key,
321 py_buffer_to_vector(value));
322 return;
323 }
324
325 if (py::isinstance<py::str>(value)) {
326 this->blackboard->set<std::string>(key, value.cast<std::string>());
327 return;
328 }
329
330 if (is_python_sequence_like(value)) {
331 py::sequence seq = value.cast<py::sequence>();
332
333 // Empty sequences do not carry enough type information to infer a native
334 // element type safely.
335 if (py::len(seq) == 0) {
336 this->blackboard->set<py::object>(key, value);
337 return;
338 }
339
340 // Bool must be checked before int because Python bool is also an int.
341 if (sequence_matches(seq, [](const py::handle &item) {
342 return py::isinstance<py::bool_>(item);
343 })) {
344 this->blackboard->set<BoolVector>(key, sequence_to_vector<bool>(seq));
345 return;
346 }
347
348 if (sequence_matches(seq, [](const py::handle &item) {
349 return is_python_int_like(item);
350 })) {
351 this->blackboard->set<IntVector>(key,
352 sequence_to_vector<std::int64_t>(seq));
353 return;
354 }
355
356 // Mixed int/float sequences are normalized to double.
357 if (sequence_matches(seq, [](const py::handle &item) {
358 return is_python_number_like(item);
359 })) {
360 this->blackboard->set<FloatVector>(key,
361 sequence_to_vector<double>(seq));
362 return;
363 }
364
365 if (sequence_matches(seq, [](const py::handle &item) {
366 return py::isinstance<py::str>(item);
367 })) {
368 this->blackboard->set<StringVector>(
369 key, sequence_to_vector<std::string>(seq));
370 return;
371 }
372
373 this->blackboard->set<py::object>(key, value);
374 return;
375 }
376
377 if (py::isinstance<py::dict>(value)) {
378 py::dict dict = value.cast<py::dict>();
379
380 // Only dict[str, T] with a homogeneous value type is converted into a
381 // native unordered_map.
382 if (py::len(dict) == 0 || !dict_has_only_string_keys(dict)) {
383 this->blackboard->set<py::object>(key, value);
384 return;
385 }
386
387 if (dict_values_match(dict, [](const py::handle &item) {
388 return py::isinstance<py::bool_>(item);
389 })) {
390 this->blackboard->set<BoolDict>(key, dict_to_unordered_map<bool>(dict));
391 return;
392 }
393
394 if (dict_values_match(dict, [](const py::handle &item) {
395 return is_python_int_like(item);
396 })) {
397 this->blackboard->set<IntDict>(
398 key, dict_to_unordered_map<std::int64_t>(dict));
399 return;
400 }
401
402 // Mixed int/float dictionaries are normalized to double.
403 if (dict_values_match(dict, [](const py::handle &item) {
404 return is_python_number_like(item);
405 })) {
406 this->blackboard->set<FloatDict>(key,
407 dict_to_unordered_map<double>(dict));
408 return;
409 }
410
411 if (dict_values_match(dict, [](const py::handle &item) {
412 return py::isinstance<py::str>(item);
413 })) {
414 this->blackboard->set<StringDict>(
415 key, dict_to_unordered_map<std::string>(dict));
416 return;
417 }
418
419 this->blackboard->set<py::object>(key, value);
420 return;
421 }
422
423 this->blackboard->set<py::object>(key, value);
424 }
