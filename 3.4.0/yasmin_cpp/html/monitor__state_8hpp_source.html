103 MonitorState(const rclcpp::Node::SharedPtr &node, std::string topic_name,
104 std::set<std::string> outcomes, MonitorHandler monitor_handler,
105 rclcpp::QoS qos = 10,
106 rclcpp::CallbackGroup::SharedPtr callback_group = nullptr,
107 int msg_queue = 10, int timeout = -1)
108 : State({}), topic_name(topic_name), monitor_handler(monitor_handler),

111 // set outcomes
112 if (timeout > 0) {

136 this->topic_name, this->qos,
137 std::bind(&MonitorState::callback, this, _1), this->options);

147 std::string
148 execute(std::shared_ptr<yasmin::blackboard::Blackboard> blackboard) override {
149
150 while (this->msg_list.empty()) {
151 std::unique_lock<std::mutex> lock(this->msg_mutex);
152 auto status =
153 this->msg_cond.wait_for(lock, std::chrono::seconds(this->timeout));
154
155 if (this->is_canceled()) {
156 return basic_outcomes::CANCEL;
157 }
158
159 if (this->timeout > 0 && status == std::cv_status::timeout) {
160 YASMIN_LOG_WARN("Timeout reached, topic '%s' is not available",
161 this->topic_name.c_str());
162 return basic_outcomes::TIMEOUT;
163 }
164 }
165
166 YASMIN_LOG_INFO("Processing msg from topic '%s'", this->topic_name.c_str());
167 std::string outcome =
168 this->monitor_handler(blackboard, this->msg_list.at(0));
169 this->msg_list.erase(this->msg_list.begin());
170
171 return outcome;
172 }
