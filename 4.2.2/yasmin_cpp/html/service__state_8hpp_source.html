189 ServiceState(const rclcpp::Node::SharedPtr &node, const std::string &srv_name,
190 CreateRequestHandler create_request_handler,
191 const yasmin::Outcomes &outcomes,
192 ResponseHandler response_handler,
193 rclcpp::CallbackGroup::SharedPtr callback_group,
194 int wait_timeout = -1, int response_timeout = -1,
195 int maximum_retry = 3)
196 : State({basic_outcomes::SUCCEED, basic_outcomes::ABORT}),
197 srv_name(srv_name), wait_timeout(wait_timeout),
198 response_timeout(response_timeout), maximum_retry(maximum_retry) {
199
200 if (this->wait_timeout > 0 || this->response_timeout > 0) {
201 this->outcomes.insert(basic_outcomes::TIMEOUT);
202 }
203
204 if (!outcomes.empty()) {
205 this->outcomes.insert(outcomes.begin(), outcomes.end());
206 }
207
208 // Assign the appropriate ROS 2 node
209 if (node == nullptr) {
210 this->node_ = YasminNode::get_instance();
211 } else {
212 this->node_ = node;
213 }
214
215 // Create a service client (compatible with different rclcpp versions)
216 this->service_client =
217 ROSClientsCache::get_or_create_service_client<ServiceT>(
218 this->node_, srv_name, callback_group);
219
220 // Set the request and response handlers
221 this->create_request_handler = create_request_handler;
222 this->response_handler = response_handler;
223
224 // Validate request handler
225 if (this->create_request_handler == nullptr) {
226 throw std::invalid_argument("create_request_handler is needed");
227 }
228 }

241 std::string execute(yasmin::Blackboard::SharedPtr blackboard) override {
242 Request request = this->create_request(blackboard);
243 std::unique_lock<std::mutex> lock(this->response_done_mutex);
244 int retry_count = 0;
245
246 // Wait for the service to become available
247 YASMIN_LOG_INFO("Waiting for service '%s'", this->srv_name.c_str());
248
249 while (!this->service_client->wait_for_service(
250 std::chrono::duration<int64_t, std::ratio<1>>(this->wait_timeout))) {
251 YASMIN_LOG_WARN("Timeout reached, service '%s' is not available",
252 this->srv_name.c_str());
253 if (retry_count < this->maximum_retry) {
254 retry_count++;
255 YASMIN_LOG_WARN("Retrying to connect to service '%s' "
256 "(%d/%d)",
257 this->srv_name.c_str(), retry_count,
258 this->maximum_retry);
259 } else {
260 return basic_outcomes::TIMEOUT;
261 }
262 }
263
264 // Send the service request
265 YASMIN_LOG_INFO("Sending request to service '%s'", this->srv_name.c_str());
266
267 // Send request with callback
268 this->service_client->async_send_request(
269 request, std::bind(&ServiceState::response_callback, this, _1));
270
271 // Wait for response with timeout
272 if (this->response_timeout > 0) {
273 while (this->response_done_cond.wait_for(
274 lock, std::chrono::seconds(this->response_timeout)) ==
275 std::cv_status::timeout) {
276 YASMIN_LOG_WARN(
277 "Timeout reached while waiting for response from service '%s'",
278 this->srv_name.c_str());
279 if (retry_count < this->maximum_retry) {
280 retry_count++;
281 YASMIN_LOG_WARN("Retrying to wait for service '%s' response (%d/%d)",
282 this->srv_name.c_str(), retry_count,
283 this->maximum_retry);
284 } else {
285 return basic_outcomes::TIMEOUT;
286 }
287 }
288 } else {
289 this->response_done_cond.wait(lock);
290 }
291
292 if (this->is_canceled()) {
293 return basic_outcomes::CANCEL;
294 }
295
296 // Process the service response
297 if (this->service_response) {
298 if (response_handler != nullptr) {
299 std::string outcome =
300 this->response_handler(blackboard, this->service_response);
301 return outcome;
302 }
303 return basic_outcomes::SUCCEED;
304 } else {
305 return basic_outcomes::ABORT;
306 }
307 }
