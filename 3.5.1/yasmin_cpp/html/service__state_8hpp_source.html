168 ServiceState(const rclcpp::Node::SharedPtr &node, std::string srv_name,
169 CreateRequestHandler create_request_handler,
170 std::set<std::string> outcomes, ResponseHandler response_handler,
171 rclcpp::CallbackGroup::SharedPtr callback_group,
172 int wait_timeout = -1, int response_timeout = -1,
173 int maximum_retry = 3)
174 : State({basic_outcomes::SUCCEED, basic_outcomes::ABORT}),
175 srv_name(srv_name), wait_timeout(wait_timeout),
176 response_timeout(response_timeout), maximum_retry(maximum_retry) {
177
178 if (this->wait_timeout > 0 || this->response_timeout > 0) {
179 this->outcomes.insert(basic_outcomes::TIMEOUT);
180 }
181
182 if (!outcomes.empty()) {
183 this->outcomes.insert(outcomes.begin(), outcomes.end());
184 }
185
186 // Assign the appropriate ROS 2 node
187 if (node == nullptr) {
188 this->node_ = YasminNode::get_instance();
189 } else {
190 this->node_ = node;
191 }
192
193 // Create a service client (compatible with different rclcpp versions)
194 this->service_client =
195 ROSClientsCache::get_or_create_service_client<ServiceT>(
196 this->node_, srv_name, callback_group);
197
198 // Set the request and response handlers
199 this->create_request_handler = create_request_handler;
200 this->response_handler = response_handler;
201
202 // Validate request handler
203 if (this->create_request_handler == nullptr) {
204 throw std::invalid_argument("create_request_handler is needed");
205 }
206 }

219 std::string
220 execute(std::shared_ptr<yasmin::blackboard::Blackboard> blackboard) override {
221 Request request = this->create_request(blackboard);
222 std::unique_lock<std::mutex> lock(this->response_done_mutex);
223 int retry_count = 0;
224
225 // Wait for the service to become available
226 YASMIN_LOG_INFO("Waiting for service '%s'", this->srv_name.c_str());
227
228 while (!this->service_client->wait_for_service(
229 std::chrono::duration<int64_t, std::ratio<1>>(this->wait_timeout))) {
230 YASMIN_LOG_WARN("Timeout reached, service '%s' is not available",
231 this->srv_name.c_str());
232 if (retry_count < this->maximum_retry) {
233 retry_count++;
234 YASMIN_LOG_WARN("Retrying to connect to service '%s' "
235 "(%d/%d)",
236 this->srv_name.c_str(), retry_count,
237 this->maximum_retry);
238 } else {
239 return basic_outcomes::TIMEOUT;
240 }
241 }
242
243 // Send the service request
244 YASMIN_LOG_INFO("Sending request to service '%s'", this->srv_name.c_str());
245
246 // Send request with callback
247 this->service_client->async_send_request(
248 request, std::bind(&ServiceState::response_callback, this, _1));
249
250 // Wait for response with timeout
251 if (this->response_timeout > 0) {
252 while (this->response_done_cond.wait_for(
253 lock, std::chrono::seconds(this->response_timeout)) ==
254 std::cv_status::timeout) {
255 YASMIN_LOG_WARN(
256 "Timeout reached while waiting for response from service '%s'",
257 this->srv_name.c_str());
258 if (retry_count < this->maximum_retry) {
259 retry_count++;
260 YASMIN_LOG_WARN("Retrying to wait for service '%s' response (%d/%d)",
261 this->srv_name.c_str(), retry_count,
262 this->maximum_retry);
263 } else {
264 return basic_outcomes::TIMEOUT;
265 }
266 }
267 } else {
268 this->response_done_cond.wait(lock);
269 }
270
271 if (this->is_canceled()) {
272 return basic_outcomes::CANCEL;
273 }
274
275 // Process the service response
276 if (this->service_response) {
277 if (response_handler != nullptr) {
278 std::string outcome =
279 this->response_handler(blackboard, this->service_response);
280 return outcome;
281 }
282 return basic_outcomes::SUCCEED;
283 } else {
284 return basic_outcomes::ABORT;
285 }
286 }
