73 ActionState(const rclcpp::Node::SharedPtr &node, std::string action_name,
74 CreateGoalHandler create_goal_handler,
75 std::set<std::string> outcomes,
76 ResutlHandler result_handler = nullptr,
77 FeedbackHandler feedback_handler = nullptr, int timeout = -1.0)
78
79 : State({}), action_name(action_name),
80 create_goal_handler(create_goal_handler),
81 result_handler(result_handler), feedback_handler(feedback_handler),
82 timeout(timeout) {
83
84 this->outcomes = {basic_outcomes::SUCCEED, basic_outcomes::ABORT,
85 basic_outcomes::CANCEL};
86
87 if (outcomes.size() > 0) {
88 for (std::string outcome : outcomes) {
89 this->outcomes.insert(outcome);
90 }
91 }
92
93 if (node == nullptr) {
94 this->node_ = YasminNode::get_instance();
95 } else {
96 this->node_ = node;
97 }
98
99 this->action_client =
100 rclcpp_action::create_client<ActionT>(this->node_, action_name);
101
102 if (this->create_goal_handler == nullptr) {
103 throw std::invalid_argument("create_goal_handler is needed");
104 }
105 }

107 void cancel_state() {
108 std::lock_guard<std::mutex> lock(this->goal_handle_mutex);
109
110 if (this->goal_handle) {
111 this->action_client->async_cancel_goal(
112 this->goal_handle, std::bind(&ActionState::cancel_done, this));
113
114 std::unique_lock<std::mutex> lock(this->action_cancel_mutex);
115 this->action_cancel_cond.wait(lock);
116 }
117
118 yasmin::State::cancel_state();
119 }
120
121 void cancel_done() { this->action_cancel_cond.notify_all(); }

123 std::string
124 execute(std::shared_ptr<yasmin::blackboard::Blackboard> blackboard) {
125
126 std::unique_lock<std::mutex> lock(this->action_done_mutex);
127
128 Goal goal = this->create_goal_handler(blackboard);
129
130 // wait for server
131 RCLCPP_INFO(this->node_->get_logger(), "Waiting for action '%s'",
132 this->action_name.c_str());
133 bool act_available = this->action_client->wait_for_action_server(
134 std::chrono::duration<int64_t, std::ratio<1>>(this->timeout));
135
136 if (!act_available) {
137 RCLCPP_WARN(this->node_->get_logger(),
138 "Timeout reached, action '%s' is not available",
139 this->action_name.c_str());
140 return basic_outcomes::TIMEOUT;
141 }
142
143 // options
144 auto send_goal_options = SendGoalOptions();
145 send_goal_options.goal_response_callback =
146 std::bind(&ActionState::goal_response_callback, this, _1);
147
148 send_goal_options.result_callback =
149 std::bind(&ActionState::result_callback, this, _1);
150
151 if (this->feedback_handler != nullptr) {
152 send_goal_options.feedback_callback =
153 [this, blackboard](typename GoalHandle::SharedPtr,
154 std::shared_ptr<const Feedback> feedback) {
155 this->feedback_handler(blackboard, feedback);
156 };
157 ;
158 }
159
160 RCLCPP_INFO(this->node_->get_logger(), "Sending goal to action '%s'",
161 this->action_name.c_str());
162 this->action_client->async_send_goal(goal, send_goal_options);
163
164 // wait for results
165 this->action_done_cond.wait(lock);
166
167 if (this->action_status == rclcpp_action::ResultCode::CANCELED) {
168 return basic_outcomes::CANCEL;
169
170 } else if (this->action_status == rclcpp_action::ResultCode::ABORTED) {
171 return basic_outcomes::ABORT;
172
173 } else if (this->action_status == rclcpp_action::ResultCode::SUCCEEDED) {
174
175 if (this->result_handler != nullptr) {
176 return this->result_handler(blackboard, this->action_result);
177 }
178
179 return basic_outcomes::SUCCEED;
180 }
181
182 return basic_outcomes::ABORT;
183 }
