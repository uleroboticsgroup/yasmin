55 MonitorState(const rclcpp::Node::SharedPtr &node, std::string topic_name,
56 std::set<std::string> outcomes, MonitorHandler monitor_handler,
57 rclcpp::QoS qos, int msg_queue, int timeout)
58 : State({}), topic_name(topic_name), monitor_handler(monitor_handler),
59 msg_queue(msg_queue), timeout(timeout), monitoring(false),
60 time_to_wait(1000) {
61
62 // set outcomes
63 if (timeout > 0) {
64 this->outcomes = {basic_outcomes::TIMEOUT};
65 } else {
66 this->outcomes = {};
67 }
68
69 if (outcomes.size() > 0) {
70 for (std::string outcome : outcomes) {
71 this->outcomes.insert(outcome);
72 }
73 }
74
75 if (node == nullptr) {
76 this->node_ = YasminNode::get_instance();
77 } else {
78 this->node_ = node;
79 }
80
81 // create subscriber
82 this->sub = this->node_->template create_subscription<MsgT>(
83 topic_name, qos, std::bind(&MonitorState::callback, this, _1));
84 }
85
86 std::string
87 execute(std::shared_ptr<yasmin::blackboard::Blackboard> blackboard) override {
88
89 float elapsed_time = 0;
90 this->msg_list.clear();
91 this->monitoring = true;
92
93 while (this->msg_list.empty()) {
94 std::this_thread::sleep_for(
95 std::chrono::microseconds(this->time_to_wait));
96
97 if (this->timeout > 0) {
98
99 if (elapsed_time / 1e6 >= this->timeout) {
100 this->monitoring = false;
101 RCLCPP_WARN(this->node_->get_logger(),
102 "Timeout reached, topic '%s' is not available",
103 this->topic_name.c_str());
104 return basic_outcomes::TIMEOUT;
105 }
106
107 elapsed_time += this->time_to_wait;
108 }
109 }
110
111 RCLCPP_INFO(this->node_->get_logger(), "Processing msg from topic '%s'",
112 this->topic_name.c_str());
113 std::string outcome =
114 this->monitor_handler(blackboard, this->msg_list.at(0));
115 this->msg_list.erase(this->msg_list.begin());
116
117 this->monitoring = false;
118 return outcome;
119 }
120
121private:
122 rclcpp::Node::SharedPtr node_;

133 void callback(const typename MsgT::SharedPtr msg) {
134
135 if (this->monitoring) {
136 this->msg_list.push_back(msg);
137
138 if ((int)this->msg_list.size() >= this->msg_queue) {
139 this->msg_list.erase(this->msg_list.begin());
140 }
141 }
142 }
