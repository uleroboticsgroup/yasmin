261 void cancel_state() override {
262
263 bool waiting_for_cancel = false;
264
265 {
266 std::lock_guard<std::mutex> lock(this->goal_handle_mutex);
267
268 if (this->goal_handle) {
269 auto future_cancel = this->action_client->async_cancel_goal(
270 this->goal_handle, std::bind(&ActionState::cancel_done, this));
271 waiting_for_cancel = true;
272 }
273 }
274
275 // Wait for cancellation to complete outside of the goal_handle_mutex lock
276 if (waiting_for_cancel) {
277 std::unique_lock<std::mutex> cancel_lock(this->action_cancel_mutex);
278 this->action_cancel_cond.wait(cancel_lock);
279 }
280
281 // Wake up the execute() method if it's waiting
282 yasmin::State::cancel_state();
283 }

309 std::string execute(yasmin::Blackboard::SharedPtr blackboard) {
310
311 std::unique_lock<std::mutex> lock(this->action_done_mutex);
312 int retry_count = 0;
313
314 Goal goal = this->create_goal_handler(blackboard);
315
316 // Wait for the action server to be available
317 YASMIN_LOG_INFO("Waiting for action '%s'", this->action_name.c_str());
318
319 while (!this->action_client->wait_for_action_server(
320 std::chrono::duration<int64_t, std::ratio<1>>(this->wait_timeout))) {
321
322 if (this->is_canceled()) {
323 return basic_outcomes::CANCEL;
324 }
325
326 YASMIN_LOG_WARN("Timeout reached, action '%s' is not available",
327 this->action_name.c_str());
328 if (retry_count < this->maximum_retry) {
329 retry_count++;
330 YASMIN_LOG_WARN("Retrying to connect to action '%s' "
331 "(%d/%d)",
332 this->action_name.c_str(), retry_count,
333 this->maximum_retry);
334 } else {
335 return basic_outcomes::TIMEOUT;
336 }
337 }
338
339 if (this->is_canceled()) {
340 return basic_outcomes::CANCEL;
341 }
342
343 // Prepare options for sending the goal
344 SendGoalOptions send_goal_options;
345 send_goal_options.goal_response_callback =
346 std::bind(&ActionState::goal_response_callback, this, _1);
347 send_goal_options.result_callback =
348 std::bind(&ActionState::result_callback, this, _1);
349
350 if (this->feedback_handler) {
351 send_goal_options.feedback_callback =
352 [this, blackboard](typename GoalHandle::SharedPtr,
353 std::shared_ptr<const Feedback> feedback) {
354 this->feedback_handler(blackboard, feedback);
355 };
356 }
357
358 // Send the goal to the action server
359 YASMIN_LOG_INFO("Sending goal to action '%s'", this->action_name.c_str());
360 this->action_client->async_send_goal(goal, send_goal_options);
361
362 if (this->response_timeout > 0) {
363 while (this->action_done_cond.wait_for(
364 lock, std::chrono::seconds(this->response_timeout)) ==
365 std::cv_status::timeout) {
366
367 if (this->is_canceled()) {
368 return basic_outcomes::CANCEL;
369 }
370
371 YASMIN_LOG_WARN(
372 "Timeout reached while waiting for response from action '%s'",
373 this->action_name.c_str());
374 if (retry_count < this->maximum_retry) {
375 retry_count++;
376 YASMIN_LOG_WARN("Retrying to wait for action '%s' response (%d/%d)",
377 this->action_name.c_str(), retry_count,
378 this->maximum_retry);
379 } else {
380 return basic_outcomes::TIMEOUT;
381 }
382 }
383
384 } else {
385 this->action_done_cond.wait(lock);
386 }
387
388 if (this->is_canceled()) {
389 return basic_outcomes::CANCEL;
390 }
391
392 switch (this->action_status) {
393 case rclcpp_action::ResultCode::CANCELED:
394 return basic_outcomes::CANCEL;
395
396 case rclcpp_action::ResultCode::ABORTED:
397 return basic_outcomes::ABORT;
398
399 case rclcpp_action::ResultCode::SUCCEEDED:
400 if (this->result_handler) {
401 return this->result_handler(blackboard, this->action_result);
402 }
403 return basic_outcomes::SUCCEED;
404
405 default:
406 return basic_outcomes::ABORT;
407 }
408 }
