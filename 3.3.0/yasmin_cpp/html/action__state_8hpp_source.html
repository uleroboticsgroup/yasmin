264 std::string
265 execute(std::shared_ptr<yasmin::blackboard::Blackboard> blackboard) {
266
267 std::unique_lock<std::mutex> lock(this->action_done_mutex);
268
269 Goal goal = this->create_goal_handler(blackboard);
270
271 // Wait for the action server to be available
272 YASMIN_LOG_INFO("Waiting for action '%s'", this->action_name.c_str());
273 bool act_available = this->action_client->wait_for_action_server(
274 std::chrono::duration<int64_t, std::ratio<1>>(this->timeout));
275
276 if (!act_available) {
277 YASMIN_LOG_WARN("Timeout reached, action '%s' is not available",
278 this->action_name.c_str());
279 return basic_outcomes::TIMEOUT;
280 }
281
282 // Prepare options for sending the goal
283 SendGoalOptions send_goal_options;
284 send_goal_options.goal_response_callback =
285 std::bind(&ActionState::goal_response_callback, this, _1);
286
287 send_goal_options.result_callback =
288 std::bind(&ActionState::result_callback, this, _1);
289
290 if (this->feedback_handler) {
291 send_goal_options.feedback_callback =
292 [this, blackboard](typename GoalHandle::SharedPtr,
293 std::shared_ptr<const Feedback> feedback) {
294 this->feedback_handler(blackboard, feedback);
295 };
296 }
297
298 YASMIN_LOG_INFO("Sending goal to action '%s'", this->action_name.c_str());
299 this->action_client->async_send_goal(goal, send_goal_options);
300
301 // Wait for the action to complete
302 this->action_done_cond.wait(lock);
303
304 switch (this->action_status) {
305 case rclcpp_action::ResultCode::CANCELED:
306 return basic_outcomes::CANCEL;
307
308 case rclcpp_action::ResultCode::ABORTED:
309 return basic_outcomes::ABORT;
310
311 case rclcpp_action::ResultCode::SUCCEEDED:
312 if (this->result_handler) {
313 return this->result_handler(blackboard, this->action_result);
314 }
315 return basic_outcomes::SUCCEED;
316
317 default:
318 return basic_outcomes::ABORT;
319 }
320 }
