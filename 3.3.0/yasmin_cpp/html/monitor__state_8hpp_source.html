102 MonitorState(const rclcpp::Node::SharedPtr &node, std::string topic_name,
103 std::set<std::string> outcomes, MonitorHandler monitor_handler,
104 rclcpp::QoS qos = 10,

141 std::string
142 execute(std::shared_ptr<yasmin::blackboard::Blackboard> blackboard) override {
143
144 float elapsed_time = 0;
145 this->msg_list.clear();
146
147 // Crate subscription
148 this->sub = this->node_->template create_subscription<MsgT>(
149 this->topic_name, this->qos,
150 std::bind(&MonitorState::callback, this, _1), this->options);
151
152 while (this->msg_list.empty()) {
153 std::this_thread::sleep_for(
154 std::chrono::microseconds(this->time_to_wait));
155
156 if (is_canceled()) {
157 this->sub.reset();
158 this->sub = nullptr;
159 return basic_outcomes::CANCEL;
160 }
161
162 if (this->timeout > 0) {
163
164 if (elapsed_time / 1e6 >= this->timeout) {
165 YASMIN_LOG_WARN("Timeout reached, topic '%s' is not available",
166 this->topic_name.c_str());
167 this->sub.reset();
168 this->sub = nullptr;
169 return basic_outcomes::TIMEOUT;
170 }
171
172 elapsed_time += this->time_to_wait;
173 }
174 }
175
176 YASMIN_LOG_INFO("Processing msg from topic '%s'", this->topic_name.c_str());
177 std::string outcome =
178 this->monitor_handler(blackboard, this->msg_list.at(0));
179 this->msg_list.erase(this->msg_list.begin());
180
181 this->sub.reset();
182 this->sub = nullptr;
183 return outcome;
184 }
