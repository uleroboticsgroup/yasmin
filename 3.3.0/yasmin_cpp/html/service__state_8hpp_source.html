122 ServiceState(const rclcpp::Node::SharedPtr &node, std::string srv_name,
123 CreateRequestHandler create_request_handler,
124 std::set<std::string> outcomes, ResponseHandler response_handler,
125 rclcpp::CallbackGroup::SharedPtr callback_group = nullptr,
126 int timeout = -1.0)
127 : State({}), srv_name(srv_name), timeout(timeout) {
128
129 this->outcomes = {basic_outcomes::SUCCEED, basic_outcomes::ABORT};
130
131 if (this->timeout >= 0) {
132 this->outcomes.insert(basic_outcomes::TIMEOUT);
133 }
134
135 if (!outcomes.empty()) {
136 this->outcomes.insert(outcomes.begin(), outcomes.end());
137 }
138
139 // Assign the appropriate ROS 2 node
140 if (node == nullptr) {
141 this->node_ = YasminNode::get_instance();
142 } else {
143 this->node_ = node;
144 }
145
146 // Create a service client (compatible with different rclcpp versions)
147#if __has_include("rclcpp/version.h")
148#include "rclcpp/version.h"
149#if RCLCPP_VERSION_GTE(28, 1, 9)
150 auto qos = rclcpp::QoS(
151 rclcpp::QoSInitialization::from_rmw(rmw_qos_profile_services_default));
152#else
153 auto qos = rmw_qos_profile_services_default;
154#endif
155#else
156 auto qos = rmw_qos_profile_services_default;
157#endif
158
159 this->service_client = this->node_->template create_client<ServiceT>(
160 srv_name, qos, callback_group);
161
162 // Set the request and response handlers
163 this->create_request_handler = create_request_handler;
164 this->response_handler = response_handler;
165
166 // Validate request handler
167 if (this->create_request_handler == nullptr) {
168 throw std::invalid_argument("create_request_handler is needed");
169 }

183 std::string
184 execute(std::shared_ptr<yasmin::blackboard::Blackboard> blackboard) override {
185
186 Request request = this->create_request(blackboard);
187
188 // Wait for the service to become available
189 YASMIN_LOG_INFO("Waiting for service '%s'", this->srv_name.c_str());
190 bool srv_available = this->service_client->wait_for_service(
191 std::chrono::duration<int64_t, std::ratio<1>>(this->timeout));
192
193 if (!srv_available) {
194 YASMIN_LOG_WARN("Timeout reached, service '%s' is not available",
195 this->srv_name.c_str());
196 return basic_outcomes::TIMEOUT;
197 }
198
199 // Send the service request
200 YASMIN_LOG_INFO("Sending request to service '%s'", this->srv_name.c_str());
201 auto future = this->service_client->async_send_request(request);
202
203 // Wait for the response
204 future.wait();
205 Response response = future.get();
206
207 if (response) {
208 if (response_handler != nullptr) {
209 std::string outcome = this->response_handler(blackboard, response);
210 return outcome;
211 }
212 return basic_outcomes::SUCCEED;
213 } else {
214 return basic_outcomes::ABORT;
215 }
216 }
217

220 rclcpp::Node::SharedPtr node_;
222 std::shared_ptr<rclcpp::Client<ServiceT>> service_client;

226 ResponseHandler response_handler;
228 std::string srv_name;
230 int timeout;

239 Request
240 create_request(std::shared_ptr<yasmin::blackboard::Blackboard> blackboard) {
241 return this->create_request_handler(blackboard);
242 }
