51inline std::vector<uint8_t> py_buffer_to_vector(const py::handle &value) {
52 char *data = nullptr;
53 py::ssize_t size = 0;
54
55 if (PyBytes_Check(value.ptr()) != 0) {
56 if (PyBytes_AsStringAndSize(value.ptr(), &data, &size) != 0) {
57 throw py::error_already_set();
58 }
59 } else if (PyByteArray_Check(value.ptr()) != 0) {
60 data = PyByteArray_AsString(value.ptr());
61 size = PyByteArray_Size(value.ptr());
62 } else {
63 throw std::runtime_error("Expected bytes or bytearray object");
64 }
65
66 return std::vector<uint8_t>(reinterpret_cast<uint8_t *>(data),
67 reinterpret_cast<uint8_t *>(data) + size);
68}

96template <typename Predicate>
97inline bool sequence_matches(const py::sequence &seq, Predicate pred) {
98 for (auto item : seq) {
99 if (!pred(item)) {
100 return false;
101 }

106template <typename T>
107inline std::vector<T> sequence_to_vector(const py::sequence &seq) {
108 std::vector<T> result;
109 result.reserve(static_cast<std::size_t>(py::len(seq)));
110
111 for (auto item : seq) {
112 result.push_back(py::cast<T>(item));
113 }
114
115 return result;
116}

118inline bool dict_has_only_string_keys(const py::dict &dict) {
119 for (auto item : dict) {
120 if (!py::isinstance<py::str>(item.first)) {
121 return false;
122 }
123 }
124 return true;

127template <typename Predicate>
128inline bool dict_values_match(const py::dict &dict, Predicate pred) {
129 for (auto item : dict) {
130 if (!pred(item.second)) {
131 return false;
132 }
133 }
134 return true;
135}

137template <typename T>
138inline std::unordered_map<std::string, T>
139dict_to_unordered_map(const py::dict &dict) {
140 std::unordered_map<std::string, T> result;
141 result.reserve(static_cast<std::size_t>(py::len(dict)));
142
143 for (auto item : dict) {
144 result.emplace(py::cast<std::string>(item.first), py::cast<T>(item.second));
145 }
146
147 return result;

163inline BlackboardKeyInfo
164blackboard_key_info_from_pyobject(const std::string &key_name,
165 py::object value) {
166 using namespace detail;
167
168 if (py::isinstance<py::bool_>(value)) {
169 return BlackboardKeyInfo(key_name, "", value.cast<bool>());
170 }
171
172 if (is_python_int_like(value)) {
173 return BlackboardKeyInfo(key_name, "", value.cast<std::int64_t>());
174 }
175
176 if (py::isinstance<py::float_>(value)) {
177 return BlackboardKeyInfo(key_name, "", value.cast<double>());
178 }
179
180 if (is_python_bytes_like(value)) {
181 return BlackboardKeyInfo(key_name, "", py_buffer_to_vector(value));
182 }
183
184 if (py::isinstance<py::str>(value)) {
185 return BlackboardKeyInfo(key_name, "", value.cast<std::string>());
186 }
187
188 if (yasmin::callback_signal_pyutils::is_python_callback_signal_like(value)) {
189 return BlackboardKeyInfo(
190 key_name, "",
191 yasmin::callback_signal_pyutils::cast_python_callback_signal(value));
192 }
193
194 if (is_python_sequence_like(value)) {
195 py::sequence seq = value.cast<py::sequence>();
196
197 if (py::len(seq) == 0) {
198 BlackboardKeyInfo info(key_name);
199 info.has_default = true;
200 info.default_value_type = "py::object";
201 auto stored = std::make_shared<py::object>(value);
202 info.default_value = stored;
203 info.inject_default = [stored](Blackboard &bb, const std::string &key) {
204 py::gil_scoped_acquire gil;
205 bb.set<py::object>(key, *stored);
206 };
207 return info;
208 }
209
210 if (sequence_matches(seq, [](const py::handle &item) {
211 return py::isinstance<py::bool_>(item);
212 })) {
213 return BlackboardKeyInfo(key_name, "", sequence_to_vector<bool>(seq));
214 }
215
216 if (sequence_matches(seq, [](const py::handle &item) {
217 return is_python_int_like(item);
218 })) {
219 return BlackboardKeyInfo(key_name, "",
220 sequence_to_vector<std::int64_t>(seq));
221 }
222
223 if (sequence_matches(seq, [](const py::handle &item) {
224 return is_python_number_like(item);
225 })) {
226 return BlackboardKeyInfo(key_name, "", sequence_to_vector<double>(seq));
227 }
228
229 if (sequence_matches(seq, [](const py::handle &item) {
230 return py::isinstance<py::str>(item);
231 })) {
232 return BlackboardKeyInfo(key_name, "",
233 sequence_to_vector<std::string>(seq));
234 }
235 }
236
237 if (py::isinstance<py::dict>(value)) {
238 py::dict dict = value.cast<py::dict>();
239
240 if (py::len(dict) != 0 && dict_has_only_string_keys(dict)) {
241 if (dict_values_match(dict, [](const py::handle &item) {
242 return py::isinstance<py::bool_>(item);
243 })) {
244 return BlackboardKeyInfo(key_name, "",
245 dict_to_unordered_map<bool>(dict));
246 }
247
248 if (dict_values_match(dict, [](const py::handle &item) {
249 return is_python_int_like(item);
250 })) {
251 return BlackboardKeyInfo(key_name, "",
252 dict_to_unordered_map<std::int64_t>(dict));
253 }
254
255 if (dict_values_match(dict, [](const py::handle &item) {
256 return is_python_number_like(item);
257 })) {
258 return BlackboardKeyInfo(key_name, "",
259 dict_to_unordered_map<double>(dict));
260 }
261
262 if (dict_values_match(dict, [](const py::handle &item) {
263 return py::isinstance<py::str>(item);
264 })) {
265 return BlackboardKeyInfo(key_name, "",
266 dict_to_unordered_map<std::string>(dict));
267 }
268 }
269 }
270
271 BlackboardKeyInfo info(key_name);
272 info.has_default = true;
273 info.default_value_type = "py::object";
274
275 auto stored = std::make_shared<py::object>(value);
276 info.default_value = stored;
277 info.inject_default = [stored](Blackboard &bb, const std::string &key) {
278 py::gil_scoped_acquire gil;
279 bb.set<py::object>(key, *stored);
280 };
281
282 return info;
283}

291inline py::object
292blackboard_key_info_get_py_default(const BlackboardKeyInfo &info) {
293 using namespace detail;
294
295 if (!info.has_default) {
296 return py::none();
297 }
298
299 const std::string &type = info.default_value_type;
300
301 if (is_exact_cpp_type<std::vector<uint8_t>>(type)) {
302 return byte_vector_to_py_bytes(
303 info.get_default_value<std::vector<uint8_t>>());
304 }
305 if (is_exact_cpp_type<std::vector<unsigned char>>(type)) {
306 return byte_vector_to_py_bytes(
307 info.get_default_value<std::vector<unsigned char>>());
308 }
309 if (is_exact_cpp_type<std::vector<char>>(type)) {
310 return byte_vector_to_py_bytes(info.get_default_value<std::vector<char>>());
311 }
312
313 if (is_exact_cpp_type<StringVector>(type)) {
314 return py::cast(info.get_default_value<StringVector>());
315 }
316 if (is_exact_cpp_type<IntVector>(type) ||
317 is_exact_cpp_type<std::vector<int>>(type) ||
318 is_exact_cpp_type<std::vector<long>>(type) ||
319 is_exact_cpp_type<std::vector<long long>>(type)) {
320 if (is_exact_cpp_type<IntVector>(type)) {
321 return py::cast(info.get_default_value<IntVector>());
322 }
323 if (is_exact_cpp_type<std::vector<int>>(type)) {
324 return py::cast(info.get_default_value<std::vector<int>>());
325 }
326 if (is_exact_cpp_type<std::vector<long>>(type)) {
327 return py::cast(info.get_default_value<std::vector<long>>());
328 }
329 return py::cast(info.get_default_value<std::vector<long long>>());
330 }
331 if (is_exact_cpp_type<FloatVector>(type) ||
332 is_exact_cpp_type<std::vector<float>>(type)) {
333 if (is_exact_cpp_type<FloatVector>(type)) {
334 return py::cast(info.get_default_value<FloatVector>());
335 }
336 return py::cast(info.get_default_value<std::vector<float>>());
337 }
338 if (is_exact_cpp_type<BoolVector>(type)) {
339 return py::cast(info.get_default_value<BoolVector>());
340 }
341
342 if (is_exact_cpp_type<StringDict>(type)) {
343 return py::cast(info.get_default_value<StringDict>());
344 }
345 if (is_exact_cpp_type<IntDict>(type) ||
346 is_exact_cpp_type<std::unordered_map<std::string, int>>(type) ||
347 is_exact_cpp_type<std::unordered_map<std::string, long>>(type) ||
348 is_exact_cpp_type<std::unordered_map<std::string, long long>>(type)) {
349 if (is_exact_cpp_type<IntDict>(type)) {
350 return py::cast(info.get_default_value<IntDict>());
351 }
352 if (is_exact_cpp_type<std::unordered_map<std::string, int>>(type)) {
353 return py::cast(
354 info.get_default_value<std::unordered_map<std::string, int>>());
355 }
356 if (is_exact_cpp_type<std::unordered_map<std::string, long>>(type)) {
357 return py::cast(
358 info.get_default_value<std::unordered_map<std::string, long>>());
359 }
360 return py::cast(
361 info.get_default_value<std::unordered_map<std::string, long long>>());
362 }
363 if (is_exact_cpp_type<FloatDict>(type) ||
364 is_exact_cpp_type<std::unordered_map<std::string, float>>(type)) {
365 if (is_exact_cpp_type<FloatDict>(type)) {
366 return py::cast(info.get_default_value<FloatDict>());
367 }
368 return py::cast(
369 info.get_default_value<std::unordered_map<std::string, float>>());
370 }
371 if (is_exact_cpp_type<BoolDict>(type)) {
372 return py::cast(info.get_default_value<BoolDict>());
373 }
374
375 if (is_exact_cpp_type<yasmin::CallbackSignal::SharedPtr>(type)) {
376 return py::cast(
377 info.get_default_value<yasmin::CallbackSignal::SharedPtr>());
378 }
379
380 if (is_exact_cpp_type<std::string>(type)) {
381 return py::cast(info.get_default_value<std::string>());
382 }
383 if (is_exact_cpp_type<std::int64_t>(type)) {
384 return py::cast(info.get_default_value<std::int64_t>());
385 }
386 if (is_exact_cpp_type<int>(type)) {
387 return py::cast(info.get_default_value<int>());
388 }
389 if (is_exact_cpp_type<long>(type)) {
390 return py::cast(info.get_default_value<long>());
391 }
392 if (is_exact_cpp_type<long long>(type)) {
393 return py::cast(info.get_default_value<long long>());
394 }
395 if (is_exact_cpp_type<float>(type)) {
396 return py::cast(info.get_default_value<float>());
397 }
398 if (is_exact_cpp_type<double>(type)) {
399 return py::cast(info.get_default_value<double>());
400 }
401 if (is_exact_cpp_type<bool>(type)) {
402 return py::cast(info.get_default_value<bool>());
403 }
404 if (type == "py::object") {
405 return info.get_default_value<py::object>();
406 }
407
408 throw std::runtime_error("Unsupported BlackboardKeyInfo default value type "
409 "for Python conversion: " +
410 type);
411}
