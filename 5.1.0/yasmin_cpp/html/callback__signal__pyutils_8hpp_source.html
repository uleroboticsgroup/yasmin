41inline bool is_python_callback_signal_like(const py::handle &value) {
42 if (value.is_none()) {
43 return false;
44 }
45
46 try {
47#if PYBIND11_VERSION_MAJOR > 2 || \
48 (PYBIND11_VERSION_MAJOR == 2 && PYBIND11_VERSION_MINOR >= 6)
49 py::module_ callback_signal_module =
50 py::module_::import("yasmin.callback_signal");
51#else
52 py::module callback_signal_module =
53 py::module::import("yasmin.callback_signal");
54#endif
55 py::object callback_signal_class =
56 callback_signal_module.attr("CallbackSignal");
57 const int is_instance =
58 PyObject_IsInstance(value.ptr(), callback_signal_class.ptr());
59
60 if (is_instance == 1) {
61 return true;
62 }
63
64 if (is_instance == -1) {
65 PyErr_Clear();
66 }
67 } catch (const py::error_already_set &) {
68 PyErr_Clear();
69 }
70
71 return false;
72}
