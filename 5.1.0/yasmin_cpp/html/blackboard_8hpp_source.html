41inline std::string demangle_type(const std::string &mangled_name) {
42
43 std::string name = mangled_name;
44
45#ifdef __GNUG__ // If using GCC/G++
46 int status;
47 // Demangle the name using GCC's demangling function
48 char *demangled =
49 abi::__cxa_demangle(name.c_str(), nullptr, nullptr, &status);
50 if (status == 0) {
51 name = demangled;
52 }
53 free(demangled);
54#endif
55
56 return name; // Return the demangled type name
57}

122 template <class T> void set(const std::string &name, T value) {
123
124 YASMIN_LOG_DEBUG("Setting '%s' in the blackboard", name.c_str());
125
126 std::lock_guard<std::recursive_mutex> lk(this->storage->mutex);
127
128 // Apply remapping if exists
129 const std::string &key = this->remap(name);
130 const std::string type_name = demangle_type(typeid(T).name());
131
132 auto type_it = this->storage->type_registry.find(key);
133 if (type_it != this->storage->type_registry.end() &&
134 type_it->second == type_name) {
135 // Same type: update existing value in-place (avoids allocation)
136 *(std::static_pointer_cast<T>(this->storage->values.at(key))) = value;
137 } else {
138 // New key or different type: (re)create entry
139 this->storage->values[key] = std::make_shared<T>(value);
140 this->storage->type_registry[key] = type_name;
141 }
142 }
143
151 template <class T> T get(const std::string &key) const {
152
153 YASMIN_LOG_DEBUG("Getting '%s' from the blackboard", key.c_str());
154
155 std::lock_guard<std::recursive_mutex> lk(this->storage->mutex);
156
157 // Check if the key exists
158 if (!this->contains(key)) {
159 throw std::runtime_error("Element '" + key +
160 "' does not exist in the blackboard");
161 }
162
163 // Return the value casted to the requested type
164 return *(std::static_pointer_cast<T>(
165 this->storage->values.at(this->remap(key))));
166 }
