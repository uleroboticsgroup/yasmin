99 static std::vector<uint8_t> py_buffer_to_vector(const py::handle &value) {
100 char *data = nullptr;
101 py::ssize_t size = 0;
102
103 if (PyBytes_Check(value.ptr()) != 0) {
104 if (PyBytes_AsStringAndSize(value.ptr(), &data, &size) != 0) {
105 throw py::error_already_set();
106 }
107 } else if (PyByteArray_Check(value.ptr()) != 0) {
108 data = PyByteArray_AsString(value.ptr());
109 size = PyByteArray_Size(value.ptr());
110 } else {
111 throw std::runtime_error("Expected bytes or bytearray object");
112 }
113
114 return std::vector<uint8_t>(reinterpret_cast<uint8_t *>(data),
115 reinterpret_cast<uint8_t *>(data) + size);
116 }

202 template <typename T>
203 static std::vector<T> sequence_to_vector(const py::sequence &seq) {
204 std::vector<T> result;
205 result.reserve(static_cast<std::size_t>(py::len(seq)));
206
207 for (auto item : seq) {
208 result.push_back(py::cast<T>(item));
209 }
210
211 return result;

251 template <typename T>
252 static std::unordered_map<std::string, T>
253 dict_to_unordered_map(const py::dict &dict) {
254 std::unordered_map<std::string, T> result;
255 result.reserve(static_cast<std::size_t>(py::len(dict)));
256
257 for (auto item : dict) {
258 result.emplace(py::cast<std::string>(item.first),
259 py::cast<T>(item.second));
260 }
261
262 return result;
263 }

329 void set(const std::string &key, py::object value) {
330 if (py::isinstance<py::bool_>(value)) {
331 this->blackboard->set<bool>(key, value.cast<bool>());
332 return;
333 }
334
335 if (is_python_int_like(value)) {
336 this->blackboard->set<std::int64_t>(key, value.cast<std::int64_t>());
337 return;
338 }
339
340 if (py::isinstance<py::float_>(value)) {
341 this->blackboard->set<double>(key, value.cast<double>());
342 return;
343 }
344
345 if (is_python_bytes_like(value)) {
346 this->blackboard->set<std::vector<uint8_t>>(key,
347 py_buffer_to_vector(value));
348 return;
349 }
350
351 if (py::isinstance<py::str>(value)) {
352 this->blackboard->set<std::string>(key, value.cast<std::string>());
353 return;
354 }
355
356 if (yasmin::callback_signal_pyutils::is_python_callback_signal_like(
357 value)) {
358 this->blackboard->set<yasmin::CallbackSignal::SharedPtr>(
359 key,
360 yasmin::callback_signal_pyutils::cast_python_callback_signal(value));
361 return;
362 }
363
364 if (is_python_sequence_like(value)) {
365 py::sequence seq = value.cast<py::sequence>();
366
367 // Empty sequences do not carry enough type information to infer a native
368 // element type safely.
369 if (py::len(seq) == 0) {
370 this->blackboard->set<py::object>(key, value);
371 return;
372 }
373
374 // Bool must be checked before int because Python bool is also an int.
375 if (sequence_matches(seq, [](const py::handle &item) {
376 return py::isinstance<py::bool_>(item);
377 })) {
378 this->blackboard->set<BoolVector>(key, sequence_to_vector<bool>(seq));
379 return;
380 }
381
382 if (sequence_matches(seq, [](const py::handle &item) {
383 return is_python_int_like(item);
384 })) {
385 this->blackboard->set<IntVector>(key,
386 sequence_to_vector<std::int64_t>(seq));
387 return;
388 }
389
390 // Mixed int/float sequences are normalized to double.
391 if (sequence_matches(seq, [](const py::handle &item) {
392 return is_python_number_like(item);
393 })) {
394 this->blackboard->set<FloatVector>(key,
395 sequence_to_vector<double>(seq));
396 return;
397 }
398
399 if (sequence_matches(seq, [](const py::handle &item) {
400 return py::isinstance<py::str>(item);
401 })) {
402 this->blackboard->set<StringVector>(
403 key, sequence_to_vector<std::string>(seq));
404 return;
405 }
406
407 this->blackboard->set<py::object>(key, value);
408 return;
409 }
410
411 if (py::isinstance<py::dict>(value)) {
412 py::dict dict = value.cast<py::dict>();
413
414 // Only dict[str, T] with a homogeneous value type is converted into a
415 // native unordered_map.
416 if (py::len(dict) == 0 || !dict_has_only_string_keys(dict)) {
417 this->blackboard->set<py::object>(key, value);
418 return;
419 }
420
421 if (dict_values_match(dict, [](const py::handle &item) {
422 return py::isinstance<py::bool_>(item);
423 })) {
424 this->blackboard->set<BoolDict>(key, dict_to_unordered_map<bool>(dict));
425 return;
426 }
427
428 if (dict_values_match(dict, [](const py::handle &item) {
429 return is_python_int_like(item);
430 })) {
431 this->blackboard->set<IntDict>(
432 key, dict_to_unordered_map<std::int64_t>(dict));
433 return;
434 }
435
436 // Mixed int/float dictionaries are normalized to double.
437 if (dict_values_match(dict, [](const py::handle &item) {
438 return is_python_number_like(item);
439 })) {
440 this->blackboard->set<FloatDict>(key,
441 dict_to_unordered_map<double>(dict));
442 return;
443 }
444
445 if (dict_values_match(dict, [](const py::handle &item) {
446 return py::isinstance<py::str>(item);
447 })) {
448 this->blackboard->set<StringDict>(
449 key, dict_to_unordered_map<std::string>(dict));
450 return;
451 }
452
453 this->blackboard->set<py::object>(key, value);
454 return;
455 }
456
457 this->blackboard->set<py::object>(key, value);
458 }
