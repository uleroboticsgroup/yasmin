260 void cancel_state() {
261 std::lock_guard<std::mutex> lock(this->goal_handle_mutex);
262
263 if (this->goal_handle) {
264 this->action_client->async_cancel_goal(
265 this->goal_handle, std::bind(&ActionState::cancel_done, this));
266
267 std::unique_lock<std::mutex> lock(this->action_cancel_mutex);
268 this->action_cancel_cond.wait(lock);
269 }

271 yasmin::State::cancel_state();

298 std::string
299 execute(std::shared_ptr<yasmin::blackboard::Blackboard> blackboard) {
300
301 std::unique_lock<std::mutex> lock(this->action_done_mutex);
302 int retry_count = 0;
303
304 Goal goal = this->create_goal_handler(blackboard);
305
306 // Wait for the action server to be available
307 YASMIN_LOG_INFO("Waiting for action '%s'", this->action_name.c_str());
308
309 while (!this->action_client->wait_for_action_server(
310 std::chrono::duration<int64_t, std::ratio<1>>(this->wait_timeout))) {
311 YASMIN_LOG_WARN("Timeout reached, action '%s' is not available",
312 this->action_name.c_str());
313 if (retry_count < this->maximum_retry) {
314 retry_count++;
315 YASMIN_LOG_WARN("Retrying to connect to action '%s' "
316 "(%d/%d)",
317 this->action_name.c_str(), retry_count,
318 this->maximum_retry);
319 } else {
320 return basic_outcomes::TIMEOUT;
321 }
322 }
323
324 // Prepare options for sending the goal
325 SendGoalOptions send_goal_options;
326 send_goal_options.goal_response_callback =
327 std::bind(&ActionState::goal_response_callback, this, _1);
328
329 send_goal_options.result_callback =
330 std::bind(&ActionState::result_callback, this, _1);
331
332 if (this->feedback_handler) {
333 send_goal_options.feedback_callback =
334 [this, blackboard](typename GoalHandle::SharedPtr,
335 std::shared_ptr<const Feedback> feedback) {
336 this->feedback_handler(blackboard, feedback);
337 };
338 }
339
340 YASMIN_LOG_INFO("Sending goal to action '%s'", this->action_name.c_str());
341 this->action_client->async_send_goal(goal, send_goal_options);
342
343 if (this->response_timeout > 0) {
344 while (this->action_done_cond.wait_for(
345 lock, std::chrono::seconds(this->response_timeout)) ==
346 std::cv_status::timeout) {
347 YASMIN_LOG_WARN(
348 "Timeout reached while waiting for response from action '%s'",
349 this->action_name.c_str());
350 if (retry_count < this->maximum_retry) {
351 retry_count++;
352 YASMIN_LOG_WARN("Retrying to wait for action '%s' response (%d/%d)",
353 this->action_name.c_str(), retry_count,
354 this->maximum_retry);
355 } else {
356 return basic_outcomes::TIMEOUT;
357 }
358 }
359 } else {
360 this->action_done_cond.wait(lock);
361 }
362
363 if (this->is_canceled()) {
364 return basic_outcomes::CANCEL;
365 }
366
367 switch (this->action_status) {
368 case rclcpp_action::ResultCode::CANCELED:
369 return basic_outcomes::CANCEL;
370
371 case rclcpp_action::ResultCode::ABORTED:
372 return basic_outcomes::ABORT;
373
374 case rclcpp_action::ResultCode::SUCCEEDED:
375 if (this->result_handler) {
376 return this->result_handler(blackboard, this->action_result);
377 }
378 return basic_outcomes::SUCCEED;
379
380 default:
381 return basic_outcomes::ABORT;
382 }
383 }
