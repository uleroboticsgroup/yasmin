190 ServiceState(const rclcpp::Node::SharedPtr &node, const std::string &srv_name,
191 CreateRequestHandler create_request_handler,
192 const std::set<std::string> &outcomes,
193 ResponseHandler response_handler,
194 rclcpp::CallbackGroup::SharedPtr callback_group,
195 int wait_timeout = -1, int response_timeout = -1,
196 int maximum_retry = 3)
197 : State({basic_outcomes::SUCCEED, basic_outcomes::ABORT}),
198 srv_name(srv_name), wait_timeout(wait_timeout),
199 response_timeout(response_timeout), maximum_retry(maximum_retry) {
200
201 if (this->wait_timeout > 0 || this->response_timeout > 0) {
202 this->outcomes.insert(basic_outcomes::TIMEOUT);
203 }
204
205 if (!outcomes.empty()) {
206 this->outcomes.insert(outcomes.begin(), outcomes.end());
207 }
208
209 // Assign the appropriate ROS 2 node
210 if (node == nullptr) {
211 this->node_ = YasminNode::get_instance();
212 } else {
213 this->node_ = node;
214 }
215
216 // Create a service client (compatible with different rclcpp versions)
217 this->service_client =
218 ROSClientsCache::get_or_create_service_client<ServiceT>(
219 this->node_, srv_name, callback_group);
220
221 // Set the request and response handlers
222 this->create_request_handler = create_request_handler;
223 this->response_handler = response_handler;
224
225 // Validate request handler
226 if (this->create_request_handler == nullptr) {
227 throw std::invalid_argument("create_request_handler is needed");
228 }
229 }
230
242 std::string execute(std::shared_ptr<yasmin::Blackboard> blackboard) override {
243 Request request = this->create_request(blackboard);
244 std::unique_lock<std::mutex> lock(this->response_done_mutex);
245 int retry_count = 0;
246
247 // Wait for the service to become available
248 YASMIN_LOG_INFO("Waiting for service '%s'", this->srv_name.c_str());
249
250 while (!this->service_client->wait_for_service(
251 std::chrono::duration<int64_t, std::ratio<1>>(this->wait_timeout))) {
252 YASMIN_LOG_WARN("Timeout reached, service '%s' is not available",
253 this->srv_name.c_str());
254 if (retry_count < this->maximum_retry) {
255 retry_count++;
256 YASMIN_LOG_WARN("Retrying to connect to service '%s' "
257 "(%d/%d)",
258 this->srv_name.c_str(), retry_count,
259 this->maximum_retry);
260 } else {
261 return basic_outcomes::TIMEOUT;
262 }
263 }
264
265 // Send the service request
266 YASMIN_LOG_INFO("Sending request to service '%s'", this->srv_name.c_str());
267
268 // Send request with callback
269 this->service_client->async_send_request(
270 request, std::bind(&ServiceState::response_callback, this, _1));
271
272 // Wait for response with timeout
273 if (this->response_timeout > 0) {
274 while (this->response_done_cond.wait_for(
275 lock, std::chrono::seconds(this->response_timeout)) ==
276 std::cv_status::timeout) {
277 YASMIN_LOG_WARN(
278 "Timeout reached while waiting for response from service '%s'",
279 this->srv_name.c_str());
280 if (retry_count < this->maximum_retry) {
281 retry_count++;
282 YASMIN_LOG_WARN("Retrying to wait for service '%s' response (%d/%d)",
283 this->srv_name.c_str(), retry_count,
284 this->maximum_retry);
285 } else {
286 return basic_outcomes::TIMEOUT;
287 }
288 }
289 } else {
290 this->response_done_cond.wait(lock);
291 }
292
293 if (this->is_canceled()) {
294 return basic_outcomes::CANCEL;
295 }
296
297 // Process the service response
298 if (this->service_response) {
299 if (response_handler != nullptr) {
300 std::string outcome =
301 this->response_handler(blackboard, this->service_response);
302 return outcome;
303 }
304 return basic_outcomes::SUCCEED;
305 } else {
306 return basic_outcomes::ABORT;
307 }
308 }
